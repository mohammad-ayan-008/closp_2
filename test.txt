fn bool comparitor(int* a,int* b){

fn void swap(int* a,int* b){
  int temp = *a;
  *a = *b;
  *b = temp;
}

fn int fibo(int num){
  if (num <=1){
    return num;
  } 
  return fibo(num-1) + fibo(num-2);
}
