fn int fiboiter(int n){
  if (n==0 || n==1){
    return n;
  }
  
  int a = 0;
  int b = 1;
  int next = 0;
  int i =2;

  while (i<= n){
    next = a +b;
    a = b;
    b = next;
    i++;
  }
  return b;
}

fn void swap(int* a,int* b){
  int temp = *a;
  *a = *b;
  *b = temp;
}

fn int fibo(int num){
  if (num <=1){
    return num;
  } 
  return fibo(num-1) + fibo(num-2);
