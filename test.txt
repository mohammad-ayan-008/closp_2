fn int fibo(int n){
  if(n == 0){
    return 0;
  }
  if(n == 1){
    return 1;
  }

  int a = 0;
  int b = 1;
  int next =0;
  int i =2;
  while (i<=n){
    next = a+b;
    a = b;
    b = next;
    i++;
  }
  return b;
}
